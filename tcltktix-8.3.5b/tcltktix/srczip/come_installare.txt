spacchettare i tre tar in una directory comune
per esempio: /usr/users/guag/work/tcltktix

TCL
cd /usr/users/guag/work/tcltktix/tcl8.3.5/unix
./configure --prefix=/usr/users/guag/work/tcltktix
make
make install

TK
cd /usr/users/guag/work/tcltktix/tk8.3.5/unix
./configure --prefix=/usr/users/guag/work/tcltktix
make
make install


TIX
cd /usr/users/guag/work/tcltktix/tix-8.1.4/unix
./configure --enable-shared --prefix=/usr/users/guag/work/tcltktix
cd tk8.3
./configure --enable-shared --prefix=/usr/users/guag/work/tcltktix
make
make install
cd ..
make install
